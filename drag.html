Drag me!
var dragDiv = document.getElementById('dragDiv'); var offsetX, offsetY; var isDragging = false; dragDiv.addEventListener('mousedown', function(event) { // Store the initial position offsets and set dragging flag offsetX = event.clientX - dragDiv.offsetLeft; offsetY = event.clientY - dragDiv.offsetTop; isDragging = true; dragDiv.style.cursor = 'grabbing'; }); document.addEventListener('mousemove', function(event) { if (isDragging) { // Update the div's position based on the mouse movement dragDiv.style.left = (event.clientX - offsetX) + 'px'; dragDiv.style.top = (event.clientY - offsetY) + 'px'; } }); document.addEventListener('mouseup', function(event) { // Reset dragging flag and cursor style isDragging = false; dragDiv.style.cursor = 'grab'; });
